using namespace std;

int main()
{
    char e, p[100];
    int i, key, n;
    
    cout << "Enter the Key Number : ";
    cin >> key;
    cout << "Enter the Length of Plaintext :";
    cin >> n;
    cout << "Enter the Characters of Plaintext :";
    
    for(i = 0; i < n; i++)
        cin >> p[i];	
    
    for(i = 0; i < n; i++)

        e = (p[i] - 'A' + key) % 26 + 'A';
        cout << p[i] - 'A' << " = " << p[i] << "  -------> " 
             << e << " = " << (p[i] - 'A' + key) % 26 << "\n";
        // cout << e;  
    }
  
    return 0;
}

#include <iostream>
using namespace std;

int main()
{
    string s, ch;
    int i, e, key;
    
    cout << "Enter the Key Number : ";
    cin >> key;
    cout << "Enter the Plaintext : ";
    cin >> s;
        
    for(i = 0; i < s.size(); i++)
    {
        e = (int(s[i]) + key);
        if (e > 90) e = e - 26;
        
        cout << s[i] << " = " << int(s[i]) 
             << "  -------> " << e 
             << " = " << char(e) << "\n";
             
        ch = ch + char(e);
    }
    
    cout << "\nEncrypted Message is : " << ch << "\n";
    
    return 0;
}
