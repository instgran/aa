    #output {
      margin-top: 20px;
      padding: 10px;
      background: #f0f0f0;
      border-radius: 5px;
    }
  </style>
</head>
<body>
  <h2>Cookie Manager</h2>
  <input type="text" id="cookieName" placeholder="Cookie Name" />
  <input type="text" id="cookieValue" placeholder="Cookie Value" />
  <br>
  <button onclick="setMyCookie()">Set Cookie</button>
  <button onclick="getMyCookie()">Get Cookie</button>
  <button onclick="deleteMyCookie()">Delete Cookie</button>
  <button onclick="showAllCookies()">Show All Cookies</button>

  <div id="output"></div>

  <script>
    function setCookie(name, value, days) {
      let expires = "";
      if (days) {
        const date = new Date();
        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
        expires = "; expires=" + date.toUTCString();
      }
      document.cookie = name + "=" + encodeURIComponent(value) + expires + "; path=/";
    }

    function getCookie(name) {
      const nameEQ = name + "=";
      const ca = document.cookie.split(';');
      for (let i = 0; i < ca.length; i++) {
        let c = ca[i].trim();
        if (c.indexOf(nameEQ) === 0) return decodeURIComponent(c.substring(nameEQ.length));
      }
      return null;
    }

    function deleteCookie(name) {
      document.cookie = name + "=; Max-Age=0; path=/";
    }

    function setMyCookie() {
      const name = document.getElementById("cookieName").value;
      const value = document.getElementById("cookieValue").value;
      if (name && value) {
        setCookie(name, value, 7);
        document.getElementById("output").textContent = `Cookie "${name}" set to "${value}"`;
      }
    }

    function getMyCookie() {
      const name = document.getElementById("cookieName").value;
      const value = getCookie(name);
      document.getElementById("output").textContent = value
        ? `Cookie "${name}" = "${value}"`
        : `Cookie "${name}" not found`;
    }

    function deleteMyCookie() {
      const name = document.getElementById("cookieName").value;
      deleteCookie(name);
      document.getElementById("output").textContent = `Cookie "${name}" deleted`;
    }

    function showAllCookies() {
      document.getElementById("output").textContent = `All cookies: ${document.cookie}`;
    }
  </script>
</body>
</html>
